def get_by_label(label):
    objs = App.ActiveDocument.Objects
    for obj in objs:
        if obj.Label == label:
            return obj
    return None

# Example usage:
torso = get_by_label("Torso")

    print("Torso angles:", torso.Placement.Rotation.toEuler())
else:
    print("No object with that label found")
